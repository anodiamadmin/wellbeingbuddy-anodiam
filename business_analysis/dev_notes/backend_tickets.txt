from pydantic import BaseModel, Field, field_validator
import re

class SetupPasswordRequest(BaseModel):
    # Field automatically enforces the 8-character minimum
    new_password: str = Field(min_length=8)

    @field_validator('new_password')
    @classmethod
    def validate_password_complexity(cls, v: str) -> str:
        if not re.search(r'[a-zA-Z]', v):
            raise ValueError('Password must contain at least one letter.')
        if not re.search(r'\d', v):
            raise ValueError('Password must contain at least one number.')
        return v

Objective: Provide a pristine, offline local cloud environment for developers without using production credentials.
Add azure-storage-blob to the project dependencies (e.g., requirements.txt or pyproject.toml).
Create a docker-compose.yml in the project root:
YAML
services:
  azurite:
    image: mcr.microsoft.com/azure-storage/azurite
    hostname: azurite
    ports:
      - "10000:10000"
    command: "azurite-blob --blobHost 0.0.0.0 --blobPort 10000"
